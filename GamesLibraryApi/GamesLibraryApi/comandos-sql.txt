create table usuario(

	id uuid not null primary key,
	nome_usuario varchar(100) not null,
	data_nascimento date not null,
	nacionalidade varchar(50) not null,
	nome_jogo varchar(30)
	constraint chk_nacionalidade check
	(nacionalidade in ('ALEMA', 'AMERICANA', 'ANGOLANA', 'ARGENTINA', 'AUSTRALIANA', 'BELGA', 'BRASILEIRA', 'CANADENSE', 'CHILENA',
	 'CHINESA', 'COLOMBIANA', 'COREANA', 'CUBANA', 'DINAMARQUESA', 'EGIPCIA', 'ESPANHOLA', 'FRANCESA', 'GREGA', 'HOLANDESA', 'INDIANA',
	 'INGLESA', 'ITALIANA', 'JAPONESA', 'LIBANESA', 'MEXICANA', 'MOCAMBICANA', 'NORUEGUESA', 'PARAGUAIA', 'PERUANA', 'PORTUGUESA', 'RUSSA',
	 'SUECA', 'SUICA', 'URUGUAIA', 'VENEZUELANA'))

);

create table jogos(

	id_jogo uuid not null primary key,
	nome_jogo varchar(30) not null,
	data_publicacao date not null,
	genero varchar (30) not null,
	preco numeric(5,2) not null,
	id_usuario uuid not null
	constraint chk_genero check
	(genero in ('ACAO', 'AVENTURA', 'RPG', 'FPS', 'PLATAFOMRA', 'ESTRATEGIA', 'ESPORTES', 'CORRIDA', 'PUZZLE', 'TERROR', 'SIMULACAO', 'MMO', 'INDIE'))

);
