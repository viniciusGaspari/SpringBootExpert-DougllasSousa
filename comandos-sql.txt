CREATE TABLE conta (
    id UUID NOT NULL PRIMARY KEY,
    login VARCHAR(20) NOT NULL UNIQUE,
    senha VARCHAR(300) NOT NULL,
    role_id UUID REFERENCES roles(id)
);
CREATE TABLE roles (
    id UUID NOT NULL PRIMARY KEY,
    nome VARCHAR(255) NOT NULL UNIQUE
);
